fun Pascal(n: Int) {
    for (i in 1..n) {
        for (j in 1 until (n-i+1)) {
            print(" ")
        }
        var C = 1
        for (k in 1..i) {
            print("$C ")
            C = C * (i - k) / k
        }
        
        println(" ")
    }
    
}

fun main() {
    val h = 4
    Pascal(h)
}
